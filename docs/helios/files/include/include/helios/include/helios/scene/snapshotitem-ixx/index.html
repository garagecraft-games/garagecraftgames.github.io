12 /**
13 * @brief Structure for representing a snapshot item of a renderable's object
14 * state at a specific point in time.
15 *
16 * The structure bundles a weak pointer to a Renderable and a pre-calculated
17 * worldMatrix, serving as a lightweight DTO to be used with a `Snapshot`.
18 *
19 * A SnapshotItem captures the essential information to render an item
20 * without the need to traverse the entire Scene Graph during the
21 * rendering phase.
22 * The worldMatrix is a copy of the SceneNode's worldMatrix at the time of this
23 * creation, ensuring that its value remains constant, even if the transformation
24 * of the SceneNode should change during rendering.
25 *
26 * Note that for the Renderable a weak pointer is stored to allow for checking its
27 * existence in the render pass. While internal properties of the Renderable
28 * might change while this SnapshotItem is processed, the worldMatrix is immutable
29 * and reflects the transformation state at the moment the Snapshot was taken.
30 *
31 * Thus, as a precondition, we assume that no changes to the Renderable are
32 * made while the Renderable "exists" as a weak ptr in a particular Snapshot
33 * of a scene that would lead to invalidating its rendering. We rely on the
34 * state information available within the Renderable as a reflection of its
35 * state when the Snapshot was taken.
36 *
37 * @todo we're intentionally preferring stability over performance by using
38 * weak_ptr instead of raw pointers. This leaves room for (performance) improvements
39 * in later iterations.
40 * The Renderable as a weak pointer can later on be further implemented into
41 * a pure DTO, storing only scalar values that get send directly to the underlying
42 * rendering backend (e.g. vaos, vbos etc.).
43 */
44 struct SnapshotItem {
45 private:
46 /**
47 * @brief A weak pointer to the actual Renderable object.
48 */
49 std::weak_ptr<const helios::rendering::Renderable> renderable_;
50
51 /**
52 * @brief A copy of the world transformation matrix for the Renderable.
53 */
54 const helios::math::mat4f worldMatrix_;
55
56 public:
57 /**
58 * @brief Prevent copying.
59 */
60 SnapshotItem(const SnapshotItem&)=delete;
61 SnapshotItem& operator=(const SnapshotItem&)=delete;
62
63 /**
64 * @brief Provide custom move constructor
65 * @see https://en.cppreference.com/w/cpp/language/move_constructor.html
66 */
67 SnapshotItem(SnapshotItem&&) noexcept = default;
68
69 /**
70 * @brief Constructs a new SnapshotItem with the specified renderable
71 * and the worldMatrix.
72 *
73 * @param renderable A weak ptr to the Renderable.
74 * @param worldMatrix A const ref to the world transformation matrix, which gets copied.
75 */
76 SnapshotItem(
77 std::weak_ptr<const helios::rendering::Renderable> renderable,
78 const helios::math::mat4f& worldMatrix
79 );
80
81 /**
82 * @brief Returns a weak ptr to the Renderable of this SnapshotItem.
83 * Use .lock() on the returned weak pointer to obtain a shared pointer for
84 * the Renderable, if it still exists. If the Renderable does not exist,
85 * the SnapshotItem should be ignored for further processing.
86 *
87 * @return A weak ptr of Renderable of this SnapshotItem.
88 */
89 [[nodiscard]] std::weak_ptr<const helios::rendering::Renderable> renderable() const noexcept;
90
91 /**
92 * @brief Returns a const ref to the worldMatrix of this SnapshotItem.
93 * The world transformation matrix is a copy from the world transformation matrix
94 * created with this SnapshotItem.
95 *
96 * @return The const ref to the world matrix of this SnapshotItem.
97 */
98 [[nodiscard]] const helios::math::mat4f& worldMatrix() const noexcept;
99 };

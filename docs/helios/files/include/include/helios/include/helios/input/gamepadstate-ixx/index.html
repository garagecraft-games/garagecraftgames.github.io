13 /**
14 * @brief A lightweight class for transferring the state of a Gamepad.
15 *
16 * This class models a reusable object representing the input state of a gamepad.
17 * The input state of a gamepad consists of the state of the left and right trigger,
18 * ranging from [0, 1], whereas 1 means fully pressed, 0 means not pressed at all.
19 *
20 * Similarly, a GamepadState provides information about the left and the right
21 * gamepad axis. Both the x- as well as the y-axis range from [-1, 1], whereas, for
22 * the x-axis, the following holds:
23 * - 0 => not moved at all.
24 * - -1 => moved all the way to the left
25 * - 1 => moved all the way to the right
26 *
27 * Analogously for the y-axis, whereas the up-position is positive, the
28 * down position is negative.
29 *
30 * Updating the values is done by calling `updateAxes` on this object
31 *
32 * For convenient access of the axis in 2D coordinates, the class provides accessors
33 * to the sticks' axes in vec2f-form.
34 *
35 * This implementation is heavily inspired by the glfw-implementation
36 * of the gamepad input state.
37 * @see https://www.glfw.org/docs/latest/input_guide.html#joystick
38 *
39 * @Note
40 * It should be emphasized that implementing APIs must account for
41 * joystick **drift**: Achieving 0 for both the x- as well as the y-axis
42 * when no (human) input occurred is probably rare.
43 */
44 class GamepadState {

102 void update() const noexcept {
103 left_ = helios::math::vec2f(axisLeftX_, axisLeftY_);
104 right_ = helios::math::vec2f(axisRightX_, axisRightY_);

141 /**
142 * Updates the axes of this GamepadState-object.
143 *
144 * For the sticks' range [-1, 1], the following holds:
145 * -1 means moved all the way left/down, 0 means not moved at all, 1 means moved all the way right/up.
146 *
147 * For the triggers' range of [0, 1] the following holds:
148 * 0 means not moved at all, 1 means moved all the way through
149 *
150 * @param axisLeftX The value of the x-axis of the left stick, in the range [-1, 1].
151 * @param axisLeftY The value of the y-axis of the left stick, in the range [-1, 1].
152 * @param axisRightX The value of the x-axis of the right stick, in the range [-1, 1].
153 * @param axisRightY The value of the yaxis of the right stick, in the range [-1, 1].
154 * @param triggerLeft The value of the axis of the left trigger, in the range [0, 1]
155 * @param triggerRight The value of the x-axis of the right trigger, in the range [0, 1]
156 */
157 void updateAxes (
158 const float axisLeftX, const float axisLeftY,
159 const float axisRightX, const float axisRightY,
160 const float triggerLeft, const float triggerRight
161 ) noexcept{
162
163 assert(axisLeftX >= -1.0f && axisLeftX <= 1.0f && "axisLeftX is out of bounds.");
164 assert(axisLeftY >= -1.0f && axisLeftY <= 1.0f && "axisLeftY is out of bounds.");
165 assert(axisRightX >= -1.0f && axisRightX <= 1.0f && "axisRightX is out of bounds.");
166 assert(axisRightY >= -1.0f && axisRightY <= 1.0f && "axisRightY is out of bounds.");
167 assert(triggerLeft >= 0.0f && triggerLeft <= 1.0f && "triggerLeft is out of bounds.");
168 assert(triggerRight >= 0.0f && triggerRight <= 1.0f && "triggerRight is out of bounds.");
169
170 axisLeftX_ = std::clamp(axisLeftX, -1.0f, 1.0f);
171 axisLeftY_ = std::clamp(axisLeftY, -1.0f, 1.0f);
172 axisRightX_ = std::clamp(axisRightX, -1.0f, 1.0f);
173 axisRightY_ = std::clamp(axisRightY, -1.0f, 1.0f);
174 triggerLeft_ = std::clamp(triggerLeft, 0.0f, 1.0f);
175 triggerRight_ = std::clamp(triggerRight, 0.0f, 1.0f);
176
177 needsUpdate_ = true;
178 }

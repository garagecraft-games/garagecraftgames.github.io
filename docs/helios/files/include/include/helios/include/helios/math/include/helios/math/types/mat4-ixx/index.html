16 * @brief Represents a 4x4 matrix, stored in column-major order.
17 *
18 * The `mat4` struct provides a lightweight and efficient way to handle 4D
19 * matrix mathematics for numeric data types.
20 * It stores its components as type `T`. For convenient access, type aliases
21 * `mat4f`, `mat4d` and `mat4i` are available, providing float/double/integer
22 * matrix representatives.

24 * @tparam T the numeric type of the matrix components.

132 /**
133 * @brief Compares this matrix element's with the rgt matrix considering
134 * an epsilon value.
135 * Returns true if for all elements the equation |a-b| <= EPSILON
136 * holds.
137 *
138 * EPSILON is set to 0.00001
139 *
140 * @param rgt The other matix to compare with this matrix for equality.
141 *
142 * @return True is the elements of the matrices are considered equal,
143 * otherwise false.
144 *
145 * @see https://realtimecollisiondetection.net/blog/?p=89
146 *
147 * @todo account for abs (values close to zero) and rel
148 * (larger values), move EPSILON to global constant
149 */
150 constexpr bool same(const mat4<T>& rgt) const {
151
152 static const T EPSILON = static_cast<T>(0.00001);
153
154 const auto* leftPtr = value_ptr(*this);
155 const auto* rgtPtr = value_ptr(rgt);
156
157 for (int i = 0; i < 16; i++) {
158 if (std::fabs(leftPtr[i] - rgtPtr[i]) > EPSILON) {
159 return false;
160 }
161 }
162
163 return true;
164 }

167 /**
168 * @brief Strictly compares the elements of this matrix with the elements
169 * of the rgt matrix.
170 *
171 * @param rgt The right matrix to compare for equal values
172 *
173 * @return Trie if all elements are equal (==), false otherwise.
174 */
175 constexpr bool operator==(const mat4<T>& rgt) const {
176
177 const auto* leftPtr = value_ptr(*this);
178 const auto* rgtPtr = value_ptr(rgt);
179
180 for (int i = 0; i < 16; i++) {

190 /**
191 * @brief Performs matrix-multiplication with another `mat4`.
192 * This matrix is the left operand, while `m` is the right operand.
193 *
194 * @param m The right-hand side `mat4<T>` for multiplication.
195 *
196 * @return A new `mat4<T>`, representing the result of the matrix-multiplication.
197 */
198 constexpr mat4<T> operator*(const mat4<T>& m) const {
199 mat4<T> A{};
200 for (int row = 0; row < 4; row++) {
201 for (int col = 0; col < 4; col++) {
202 T sum = T{};
203 sum += (*this)(row, 0) * m(0, col);
204 sum += (*this)(row, 1) * m(1, col);
205 sum += (*this)(row, 2) * m(2, col);
206 sum += (*this)(row, 3) * m(3, col);
207 A(row, col) = sum;
208 }
209 }
210
211 return A;
212 }
213 };
